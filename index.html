BTS SIO - Services Informatiques aux Organisations

Le BTS Services Informatiques aux Organisations (SIO) est une formation qui prépare les étudiants aux métiers du développement et de la maintenance informatique. Le programme comprend 2 options :

  • SISR : Solutions d'Infrastructure, Systèmes et Réseaux
  • SLAM : Solutions Logicielles et Applications Métiers

Objectifs de la formation

Le BTS SIO a pour objectif de former des techniciens capables de :

  • Concevoir, développer et maintenir des applications informatiques
  • Installer, administrer et gérer la maintenance des équipements et des réseaux informatiques
  • Apporter un support technique aux utilisateurs
  • Intervenir sur les systèmes de gestion des entreprises

Spécifications du BTS SIO :

  • Formation de niveau III (Bac+2)
  • Durée de la formation : 2 ans
  • Alternance entre cours théoriques et stages pratiques

Programme du BTS SIO

Le programme du BTS SIO est organisé en unités d'enseignement, comprenant des matières générales et des matières professionnelles.

  • Enseignement général : Culture générale et expression, Langue vivante étrangère, Économie, Droit, Management des entreprises
  • Enseignement professionnel : Mathématiques, Analyse économique, Gestion de projet, Architecture technique, Infrastructures et réseaux, Développement d'applications

Débouchés professionnels

Les titulaires du BTS SIO peuvent occuper différents postes tels que :

  • Développeur d'applications
  • Intégrateur web
  • Technicien réseau
  • Administrateur systèmes et réseaux
  • Chargé de support technique

Ils peuvent également poursuivre leurs études en licence professionnelle, école d'ingénieur ou autre formation supérieure spécialisée.

def mrzGet():
    for image in cv_img:
        cv2.imshow("originale", image)
        cv2.waitKey(0)
        # load the image, resize it, and convert it to grayscale
        gray = cv2.cvtColor(image, cv2.COLOR_BGR2GRAY)
        cv2.imshow("gray", gray)
        cv2.waitKey(0)
        # smooth the image using a 3x3 Gaussian, then apply the blackhat
        # morphological operator to find dark regions on a light background
        gray = cv2.GaussianBlur(gray, (7, 3), 0)
        cv2.imshow("blur", gray)
        cv2.waitKey(0)
        # Applying the Black-Hat operation
        tophat_img = cv2.morphologyEx(gray, cv2.MORPH_BLACKHAT, kernel)
        cv2.imshow("blackhat", tophat_img)
        cv2.waitKey(0)
        edged = cv2.Canny(gray, 0, 255)
        cv2.imshow("canny", edged)
        cv2.waitKey(0)
        # compute the Scharr gradient of the blackhat image and scale the
        # result into the range [0, 255]
        gradX = cv2.Sobel(tophat_img, ddepth=cv2.CV_32F, dx=1, dy=0, ksize=3)
        gradX = np.absolute(gradX)
        (minVal, maxVal) = (np.min(gradX), np.max(gradX))
        gradX = (255 * ((gradX - minVal) / (maxVal - minVal))).astype("uint8")
        # apply a closing operation using the rectangular kernel to close
        # gaps in between letters -- then apply Otsu's thresholding method
        gradX = cv2.morphologyEx(gradX, cv2.MORPH_CLOSE, rectKernel)
        cv2.imshow("sobel", gradX)
        cv2.waitKey(0)
        thresh = cv2.threshold(gradX, 0, 255, cv2.THRESH_BINARY | cv2.THRESH_OTSU)[1]
        cv2.imshow("thresh", thresh)
        cv2.waitKey(0)
        # perform another closing operation, this time using the square
        # kernel to close gaps between lines of the MRZ, then perform a
        # series of erosions to break apart connected components
        thresh = cv2.erode(thresh, None, iterations=1)
        cv2.imshow("morph1", thresh)
        cv2.waitKey(0)
        thresh = cv2.morphologyEx(thresh, cv2.MORPH_DILATE, dilateKernel)
        cv2.imshow("morph2", thresh)
        cv2.waitKey(0)
        thresh = cv2.morphologyEx(thresh, cv2.MORPH_CLOSE, sqKernel)
        cv2.imshow("morph3", thresh)
        cv2.waitKey(0)
        # during thresholding, it's possible that border pixels were
        # included in the thresholding, so let's set 5% of the left and
        # right borders to zero
        p = int(image.shape[1] * 0.05)
        thresh[:, 0:p] = 0
        thresh[:, image.shape[1] - p:] = 0
        # find contours in the thresholded image and sort them by their
        # size
        cnts = cv2.findContours(thresh.copy(), cv2.RETR_EXTERNAL,
                                cv2.CHAIN_APPROX_SIMPLE)
        cnts = imutils.grab_contours(cnts)
        cnts = sorted(cnts, key=cv2.contourArea, reverse=True)
        # loop over the contours

    for c in cnts:
        # compute the bounding box of the contour and use the contour to
        # compute the aspect ratio and coverage ratio of the bounding box
        # width to the width of the image
        (x, y, w, h) = cv2.boundingRect(c)
        ar = w / float(h)
        crWidth = w / float(gray.shape[1])
        # check to see if the aspect ratio and coverage width are within
        # acceptable criteria
        if ar > 4:
            # pad the bounding box since we applied erosions and now need
            # to re-grow it
            pX = int((x + w) * 0.03)
            pY = int((y + h) * 0.03)
            (x, y) = (x - pX, y - pY)
            (w, h) = (w + (pX * 2), h + (pY * 2))
            # extract the ROI from the image and draw a bounding box
            # surrounding the MRZ
            roi = image[y:y + h, x:x + w].copy()
            cv2.rectangle(image, (x, y), (x + w, y + h), (0, 255, 0), 2)
            # show the output images
            cv2.imshow("Image", image)
            cv2.imshow("ROI", roi)
            cv2.waitKey(0)
            cv2.destroyAllWindows

using System;
using System.Collections.Generic;
using System.Linq;
using System.Windows;
using System.Windows.Controls;
using System.Windows.Media;
using System.Windows.Shapes;

namespace sneks
{
    public static class Snake
    {
        public struct Pos
        {
            int X;
            int Y;
        }

        public static List SnakeParts { get; set; } = new List();
        public static Pos FoodPos { get; set; }

        public static Grid SnakeGrid { get; set; } = new Grid();
        public static Rectangle snakeHead { get; set; } = new Rectangle() { Fill = Brushes.LightSeaGreen };
        public static List snakeBody { get; set; } = new List { new Rectangle() { Fill = Brushes.DarkKhaki } };
        public static eChosenDirection chosenDirection { get; set; } = eChosenDirection.None;
        public static Rectangle food { get; set; } = new Rectangle() { Fill = Brushes.Red };
        public static Random random { get; } = new Random((int)(DateTime.Now.Ticks % int.MaxValue));
        public static bool Playing { get; set; } = false;

        //a la place des conditions de merde a rallonge utilisées plusieurs fois peut etre utiliser
        //l'enum eSnakeGridValue et check quelle type de grid il y a a venir //autre type de grid directement



        public static void GridDesign()
        {
            SnakeGrid.Background = new SolidColorBrush(Colors.White);
            for (int i = 0; i < 16; i++)
            {
                SnakeGrid.RowDefinitions.Add(new RowDefinition() { Height = new GridLength(31.25) });
                SnakeGrid.ColumnDefinitions.Add(new ColumnDefinition() { Width = new GridLength(31.25) });

                for (int j = 0; j < 16; j++)
                {
                    Border b = new Border();
                    b.BorderBrush = new SolidColorBrush(Colors.Black);
                    b.BorderThickness = new Thickness(0.5d);
                    Grid.SetColumn(b, i);
                    Grid.SetRow(b, j);
                    SnakeGrid.Children.Add(b);
                }
            }

            Grid.SetRow(snakeHead, (int)SnakeGrid.RowDefinitions.Count() / 2);
            Grid.SetColumn(snakeHead, (int)SnakeGrid.ColumnDefinitions.Count() / 2);

            SnakeGrid.Children.Add(snakeHead);

            SnakeGrid.Children.Add(food);
            PlaceNewFood();
        }

        private static void ColorGen()
        {
            byte redH = (byte)random.Next(256);
            byte greenH = (byte)random.Next(256);
            byte blueH = (byte)random.Next(256);

            snakeHead.Fill = new SolidColorBrush(Color.FromRgb(redH, greenH, blueH));

            foreach (Rectangle parts in snakeBody)
            {
                byte redB = (byte)random.Next(256);
                byte greenB = (byte)random.Next(256);
                byte blueB = (byte)random.Next(256);

                parts.Fill = new SolidColorBrush(Color.FromRgb(redB, greenB, blueB));
            }
        }

        private static void PlaceNewFood()
        {
            do
            {
                Grid.SetRow(food, random.Next(0, SnakeGrid.RowDefinitions.Count() - 1));
                Grid.SetColumn(food, random.Next(0, SnakeGrid.ColumnDefinitions.Count() - 1));

            } while (IsFoodOnSnake(food));

        }

        private static bool IsFoodOnSnake(Rectangle food)
        {
            foreach (Rectangle bodyPart in snakeBody)
            {
                if (food.GetValue(Grid.RowProperty).Equals(bodyPart.GetValue(Grid.RowProperty)) && //check si la food est placée sur le body du snake
                    food.GetValue(Grid.ColumnProperty).Equals(bodyPart.GetValue(Grid.ColumnProperty)))
                {
                    return true;
                }
            }

            if (food.GetValue(Grid.RowProperty).Equals(snakeHead.GetValue(Grid.RowProperty)) && //check si la food est placée sur la tete du snake
                food.GetValue(Grid.ColumnProperty).Equals(snakeHead.GetValue(Grid.ColumnProperty)))
            {
                return true;
            }

            return false;
        }


        public static void NextStep()
        {
            if (!Playing)
                return;

            // current position of head
            int headRowNext = (int)snakeHead.GetValue(Grid.RowProperty);
            int headColNext = (int)snakeHead.GetValue(Grid.ColumnProperty);
            
            // next position of head
            switch (chosenDirection)
            {
                case eChosenDirection.Up:
                    headRowNext -= 1;
                    break;
                case eChosenDirection.Down:
                    headRowNext += 1;
                    break;
                case eChosenDirection.Left:
                    headColNext -= 1;
                    break;
                case eChosenDirection.Right:
                    headColNext += 1;
                    break;
            }

            // rainbow
            ColorGen();

            // next position of head is out of bounds ?
            if (headRowNext < 0 || headRowNext >= SnakeGrid.RowDefinitions.Count() || headColNext < 0 || headColNext >= SnakeGrid.ColumnDefinitions.Count())
            {
                Lose();
            }

            // next position of snake hits itself ?
            if (snakeBody.Count() > 1)
            {
                foreach (Rectangle bodyPart in snakeBody)
                {
                    if (headRowNext == (int)bodyPart.GetValue(Grid.RowProperty) && headColNext == (int)bodyPart.GetValue(Grid.ColumnProperty))
                    {
                        Lose();
                    }
                }
            }

            // is next position food ?
            Rectangle newBodyPart = null;
            if (headRowNext == (int)food.GetValue(Grid.RowProperty) && headColNext == (int)food.GetValue(Grid.ColumnProperty))
            {
                PlaceNewFood();

                newBodyPart = new Rectangle() { Fill = Brushes.DarkKhaki };

                if (snakeBody.Count() > 1)
                {
                    Grid.SetColumn(newBodyPart, (int)snakeBody[snakeBody.Count() - 1].GetValue(Grid.ColumnProperty));
                    Grid.SetRow(newBodyPart, (int)snakeBody[snakeBody.Count() - 1].GetValue(Grid.RowProperty));
                }
                else
                {
                    Grid.SetColumn(newBodyPart, (int)snakeHead.GetValue(Grid.ColumnProperty));
                    Grid.SetRow(newBodyPart, (int)snakeHead.GetValue(Grid.RowProperty));
                }

                SnakeGrid.Children.Add(newBodyPart);
            }

            // move head
            Grid.SetRow(snakeHead, headRowNext);
            Grid.SetColumn(snakeHead, headColNext);

            // move body
            if (snakeBody.Count() > 1)
            {
                for (int i = snakeBody.Count - 1; i > 0; i--)
                {
                    Grid.SetRow(snakeBody[i], (int)snakeBody[i - 1].GetValue(Grid.RowProperty));
                    Grid.SetColumn(snakeBody[i], (int)snakeBody[i - 1].GetValue(Grid.ColumnProperty));
                }
            }

            Grid.SetRow(snakeBody[0], headRowNext);
            Grid.SetColumn(snakeBody[0], headColNext);


            if (newBodyPart != null)
                snakeBody.Add(newBodyPart);

            // pas besoin de faire un snakeBody.Count - 1 car il y a la tete a compter aussi
            if (snakeBody.Count == SnakeGrid.RowDefinitions.Count * SnakeGrid.ColumnDefinitions.Count)
            {
                System.Windows.MessageBox.Show("Congratulations! You have won the game!");
                ResetStates();
            }
        }

        public static void Lose()
        {
            System.Windows.MessageBox.Show("You lost the game!");
            ResetStates();
            return;
        }

        public static void ResetStates()
        {
            SnakeGrid = new Grid();
            snakeHead = new Rectangle() { Fill = Brushes.LightSeaGreen };
            snakeBody = new List { new Rectangle() { Fill = Brushes.DarkKhaki } };
            chosenDirection = eChosenDirection.None;
            food = new Rectangle() { Fill = Brushes.Red };
            Playing = false;
        }
    }
